CREATE TABLE stores (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  store_name TEXT NOT NULL,
  region TEXT NOT NULL,      -- 서울/부산/대구/인천
  category TEXT NOT NULL,    -- 업종(커피/편의점/베이커리 등)
  monthly_sales INTEGER NOT NULL, -- 월매출(만원 단위 같은 걸로 가정)
  staff_count INTEGER NOT NULL
);

SELECT
  region,
  COUNT(*) AS store_cnt,
  AVG(monthly_sales) AS avg_sales,
  SUM(monthly_sales) AS total_sales
FROM stores
GROUP BY region
ORDER BY total_sales DESC;
